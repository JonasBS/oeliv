      if (e.target === modal) {

        modal.classList.remove('is-visible');

        modal.setAttribute('aria-hidden','true');

      }

    });


    // Toggle rates panel

    const toggleRates = document.querySelector('.js-toggle-rates');

    const ratePanel = document.querySelector('.booking-rate-panel');

      const hidden = ratePanel.hasAttribute('hidden');

      if (hidden) ratePanel.removeAttribute('hidden'); else ratePanel.setAttribute('hidden','');

    // Simple season logic for dynamic hint

    const dateInput = document.getElementById('booking-date');

    const nightsSelect = document.getElementById('booking-nights');

    const seasonLine = document.getElementById('rate-season-line');

    const minNightsLine = document.getElementById('rate-min-nights-line');


    function updateRateHint(){

      const val = dateInput.value;

      if(!val){ seasonLine.textContent = "Vælg en ankomstdato for at se sæson og prisestimat."; minNightsLine.textContent = ""; return; }

      const d = new Date(val);

      const m = d.getMonth() + 1; // 1-12

      let season = "Lavsæson";

      let price = 1200;

      let minNights = 1;


      if ((m>=10 && m<=12) || (m>=1 && m<=4)) { season = "Lavsæson"; price = 1200; minNights = 1; }

      if ((m===5) || (m===6) || (m===9)) { season = "Skuldersæson"; price = 1400; minNights = 2; }

      if (m===7 || m===8) { season = "Højsæson"; price = 1700; minNights = 3; }


      seasonLine.innerHTML = `Estimat: <strong>${season}</strong> · fra ca. <strong>${price} DKK</strong> pr. nat.`;

      const chosen = parseInt(nightsSelect.value || "1", 10);

      minNightsLine.textContent = chosen < minNights

        ? `Bemærk: Min. ${minNights} nætter i ${season.toLowerCase()}.`

        : `Valg: ${chosen} nat/ nætter opfylder min. krav i ${season.toLowerCase()}.`;

    }

    dateInput?.addEventListener('change', updateRateHint);

    nightsSelect?.addEventListener('change', updateRateHint);

      e.preventDefault();

      modal.setAttribute('aria-hidden','true');

      e.target.reset();

    });
